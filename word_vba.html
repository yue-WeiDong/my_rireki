VBA票据打印宏 - 代码说明
这是一个用于Microsoft Word的VBA宏，用于根据用户输入自动生成格式化的票据/标签文档，支持多种纸张尺寸和产品组合。

核心功能
纸张设置：支持两种自定义纸张尺寸（7.5×13cm 和 10×11.5cm）
产品组合：支持单品、双品、三品、四品等不同配置
日期管理：内置多组预设日期代码，通过用户界面选择
格式控制：自动设置字体、字号、行距、对齐方式等格式
打印控制：支持份数设置和指定打印机输出

适用于需要批量打印标准化票据、标签或产品标识的场景，特别适合多产品组合的标签打印需求。
-------------------------------------------------------------------------------------------------------------

Sub danPin() '单品格式设置

For iRow = 1 To 2
    Set myRange = ActiveDocument.Paragraphs(iRow).Range
    With myRange.Font
     .Bold = True
     .Name = "微软雅黑"
     .Size = 72
     End With
     
    With ActiveDocument.Paragraphs(iRow).Range.ParagraphFormat
    .LineSpacingRule = wdLineSpaceExactly
    .LineSpacing = 80
    .Alignment = wdAlignParagraphCenter
    End With
Next iRow

    ActiveDocument.Paragraphs(1).Range.ParagraphFormat.LineUnitBefore = 0
    ActiveDocument.Paragraphs(1).Range.ParagraphFormat.SpaceBefore = 0
End If

    .Font.Bold = False
    .Font.Superscript = wdToggle

    .ParagraphFormat.CharacterUnitFirstLineIndent = 0.5

    Set myRange = ActiveDocument.Paragraphs(iRow).Range
    With myRange.Font
     .Bold = False
     .Name = "仿宋"
     .Size = 23
     End With
     
    With ActiveDocument.Paragraphs(iRow).Range.ParagraphFormat
    .LineSpacingRule = wdLineSpaceExactly
    .LineSpacing = 36
    .Alignment = wdAlignParagraphCenter
    End With
Next iRow

End Sub

Sub sanPin() '三品格式设置
With ActiveDocument.PageSetup
.LeftMargin = CentimetersToPoints(0.5)
.RightMargin = CentimetersToPoints(0.5)
End With
  
For iRow = 1 To 4
    Set myRange = ActiveDocument.Paragraphs(iRow).Range
    With myRange.Font
     .Bold = False
     .Name = "仿宋"
     .Size = 24
     End With
     
    With ActiveDocument.Paragraphs(iRow).Range.ParagraphFormat
    .LineSpacingRule = wdLineSpaceExactly
    .LineSpacing = 60
    .Alignment = wdAlignParagraphCenter
    If iRow < 4 Then
        .BaseLineAlignment = wdBaselineAlignFarEast50
    End If
    End With
    
    If iRow < 4 Then
        With myRange.Characters(1).Font
         .Bold = True
         .Name = "微软雅黑"
         .Size = 48
         .Position = -3
        End With
        
        With myRange.Characters(2).Font
         .Size = 48
         .Superscript = wdToggle
        End With
    End If
Next iRow
End Sub
Sub siPin() '四品格式设置
With ActiveDocument.PageSetup
.LeftMargin = CentimetersToPoints(0.5)
.RightMargin = CentimetersToPoints(0.5)

For iRow = 1 To 5

     .Size = 24
     End With

    With myRange.Font
     .Size = 48
     .Superscript = wdToggle
    End With
    Next iRow

End Sub
Sub FormatText新票()
    ' 定义查找的文本
    Dim findText As String
    findText = "新"
    
    ' 查找文档中的文本
    With Selection.Find
        .Text = findText
        .Replacement.Text = ""
        .Forward = True
        .Wrap = wdFindContinue
        .Format = True
        .MatchCase = False
        .MatchWholeWord = False
        .MatchWildcards = False
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    
        ' 查找下一个匹配项
        While .Execute = True
            ' 找到的文本应用格式
    Selection.Range.ModifyEnclosure Style:=wdEncloseStyleLarge, Symbol:= _
    wdEnclosureCircle
            
            ' 查找下一个匹配项
        Wend
    End With
End Sub
